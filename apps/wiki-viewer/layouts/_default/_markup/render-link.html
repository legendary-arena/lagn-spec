{{- /* why: D-13809 default — out-of-tree links (`../**` from a wiki page) rewrite to GitHub blob URLs at build time. The wiki source paths are repo-root-relative when resolved from docs/wiki/, so `../ai/ARCHITECTURE.md` → `docs/ai/ARCHITECTURE.md` and `../../packages/foo/bar.ts` → `packages/foo/bar.ts`. Implemented in this render hook (not as a post-build HTML walk) because: (a) post-build walking adds an HTML-parser dependency and a non-determinism vector (DOM iteration order isn't guaranteed across parser versions); (b) the render hook stays inside Hugo's deterministic pipeline, so two consecutive builds produce byte-identical output (WP-139 §Determinism). Internal `slug.md` refs resolve to `//`. `INDEX.md` resolves to the home `/` (the projected `_index.md`). */ -}} {{- $dest := .Destination -}} {{- $text := .Text -}} {{- $title := "" -}} {{- with .Title -}}{{- $title = . -}}{{- end -}} {{- if or (hasPrefix $dest "http://") (hasPrefix $dest "https://") (hasPrefix $dest "mailto:") (hasPrefix $dest "#") -}} {{ $text | safeHTML }} {{- else if hasPrefix $dest "../" -}} {{- $abs := path.Join "docs/wiki" $dest -}} {{- $githubBase := "https://github.com/barefootbetters/legendary-arena/blob/main/" -}} {{ $text | safeHTML }} {{- else -}} {{- $parts := split $dest "#" -}} {{- $pathPart := index $parts 0 -}} {{- $anchor := "" -}} {{- if gt (len $parts) 1 -}} {{- $anchor = printf "#%s" (index $parts 1) -}} {{- end -}} {{- $slug := strings.TrimSuffix ".md" $pathPart -}} {{- $href := "" -}} {{- if or (eq $slug "INDEX") (eq $slug "") -}} {{- $href = "/" -}} {{- else -}} {{- $href = printf "/%s/" (lower $slug) -}} {{- end -}} {{ $text | safeHTML }} {{- end -}}
